def compare_commas(base, actual)
  bc = base.count(',')
  ac = actual.count(',')
  if (bc == ac)
    return true
  else
    return false
  end
end

def create_phoneme_array(n)
  phoneme_array = []
  n.times do
    phoneme_array << Phoneme.new
  end
  return phoneme_array
end

def get_diacritics(base)
  da = []
  ba = base.split(',')
  ba.each do |d|
    r = d.scan(/;(\.\d+)?/)
    r.each { |i| da << i } if (r != nil)
  end
  da = da.join(',')
  return da
end
